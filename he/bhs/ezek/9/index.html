1וַיִּקְרָא בְאָזְנַי קוֹל גָּדוֹל לֵאמֹר קָרְבוּ פְּקֻדּוֹת הָעִיר וְאִישׁ כְּלִי מַשְׁחֵתוֹ בְּיָדוֹ ׃ 2וְהִנֵּה שִׁשָּׁה אֲנָשִׁים בָּאִים מִדֶּרֶךְ־שַׁעַר הָעֶלְיוֹן אֲשֶׁר מָפְנֶה צָפוֹנָה וְאִישׁ כְּלִי מַפָּצוֹ בְּיָדוֹ וְאִישׁ־אֶחָד בְּתוֹכָם לָבֻשׁ בַּדִּים וְקֶסֶת הַסֹּפֵר בְּמָתְנָיו וַיָּבֹאוּ וַיַּעַמְדוּ אֵצֶל מִזְבַּח הַנְּחֹשֶׁת ׃ 3וּכְבוֹד אֱלֹהֵי יִשְׂרָאֵל נַעֲלָה מֵעַל הַכְּרוּב אֲשֶׁר הָיָה עָלָיו אֶל מִפְתַּן הַבָּיִת וַיִּקְרָא אֶל־הָאִישׁ הַלָּבֻשׁ הַבַּדִּים אֲשֶׁר קֶסֶת הַסֹּפֵר בְּמָתְנָיו ׃ ס 4וַיֹּאמֶר יְהוָה אֵלוֹ עֲבֹר בְּתוֹךְ הָעִיר בְּתוֹךְ יְרוּשָׁלִָם וְהִתְוִיתָ תָּו עַל־מִצְחוֹת הָאֲנָשִׁים הַנֶּאֱנָחִים וְהַנֶּאֱנָקִים עַל כָּל־הַתּוֹעֵבוֹת הַנַּעֲשׂוֹת בְּתוֹכָהּ ׃ 5וּלְאֵלֶּה אָמַר בְּאָזְנַי עִבְרוּ בָעִיר אַחֲרָיו וְהַכּוּ עַל־תָּחֹס עֵינְיכֶם וְאַל־תַּחְמֹלוּ ׃ 6זָקֵן בָּחוּר וּבְתוּלָה וְטַף וְנָשִׁים תַּהַרְגוּ לְמַשְׁחִית וְעַל־כָּל־אִישׁ אֲשֶׁר־עָלָיו הַתָּו אַל־תִּגַּשׁוּ וּמִמִּקְדָּשִׁי תָּחֵלּוּ וַיָּחֵלּוּ בָּאֲנָשִׁים הַזְּקֵנִים אֲשֶׁר לִפְנֵי הַבָּיִת ׃ 7וַיֹּאמֶר אֲלֵיהֶם טַמְּאוּ אֶת־הַבַּיִת וּמַלְאוּ אֶת־הַחֲצֵרוֹת חֲלָלִים צֵאוּ וְיָצְאוּ וְהִכּוּ בָעִיר ׃ 8וַיְהִי כְּהַכּוֹתָם וְנֵאשֲׁאַר אָנִי וָאֶפְּלָה עַל־פָּנַי וָאֶזְעַק וָאֹמַר אֲהָהּ אֲדֹנָי יְהוִה הֲמַשְׁחִית אַתָּה אֵת כָּל־שְׁאֵרִית יִשְׂרָאֵל בְּשָׁפְכְּךָ אֶת־חֲמָתְךָ עַל־יְרוּשָׁלִָם ׃ 9וַיֹּאמֶר אֵלַי עֲוֹן בֵּית־יִשְׂרָאֵל וִיהוּדָה גָּדוֹל בִּמְאֹד מְאֹד וַתִּמָּלֵא הָאָרֶץ דָּמִים וְהָעִיר מָלְאָה מֻטֶּה כִּי אָמְרוּ עָזַב יְהוָה אֶת־הָאָרֶץ וְאֵין יְהוָה רֹאֶה ׃ 10וְגַם־אֲנִי לֹא־תָחוֹס עֵינִי וְלֹא אֶחְמֹל דַּרְכָּם בְּרֹאשָׁם נָתָתִּי ׃ 11וְהִנֵּה הָאִישׁ לְבֻשׁ הַבַּדִּים אֲשֶׁר הַקֶּסֶת בְּמָתְנָיו מֵשִׁיב דָּבָר לֵאמֹר עָשִׂיתִי כַאֲשֶׁר צִוִּיתָנִי ׃ ס
